180 inline mln_uart(UART_t new_inst, uint32_t baud)
181 {
182 if(new_inst > MLN_UART_NUM_INSTS)
183 return;
184
185 memset(buffer, 0, sizeof(buffer));
186 index = 0;
187
188 inst = MLN_UART_TO_USART(new_inst);

190 init_pins(new_inst);

192 inst->BAUD = (uint16_t)MLN_UART_BAUD_NUM(baud);

194 inst->CTRLA = USART_RXCIE_bm;

198 init_stream();

225 /*
226 * @brief Push new read data into RX buffer of SPI class

228 inline void push(void)
229 {
230 buffer[index] = read();
231 index = (index + 1) % MLN_UART_BUF_SIZE;
232 }
233 /*
234 * @brief Pull data from RX buffer of SPI class
235 *
236 * @param new_buffer Buffer to copy data to
237 *
238 * @returns Length of data copied
239 */
240 inline uint8_t pull(uint8_t* new_buffer)
241 {
242 memccpy(new_buffer, buffer, 0, MLN_UART_BUF_SIZE);
243 memset(buffer, 0, sizeof(buffer));
244
245 uint8_t copied = index;
246 index = 0;

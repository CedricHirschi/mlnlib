96 /*
97 * @brief Configure the ADC peripheral
98 *
99 * @param standby Whether peripheral is active when device is in standby
100 * @param res Resolution to be used
101 * @param samp Number of samples to be acquired per reading
102 * @param presc The frequency prescaler to be used
103 */
104 inline const void config(bool standby = false, ADC_RES_t res = HIRES, ADC_SAMP_t samp = NONE, ADC_PRE_t presc = DIV2)
105 {
106 if((res == HIRES) && (adc->CTRLA & ADC_RESSEL_10BIT_gc)) adc->CTRLA &= ~ADC_RESSEL_10BIT_gc;
107 else adc->CTRLA |= ADC_RESSEL_10BIT_gc;
108
109 adc->CTRLA |= (uint8_t)(ADC_RUNSTBY_bm * standby);
110
111 adc->CTRLB = samp;
112
113 adc->CTRLC = presc;
114 }

161 /*
162 * @brief Read current result from ADC peripheral

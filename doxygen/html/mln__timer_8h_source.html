49uint8_t MLN_TIMER_GET_DIV(uint32_t T)
50{
51 uint8_t index = 0;
52
53 while(T > mln_max_periods[index])
54 {
55 index++;
56 if(index == sizeof(mln_timer_divs) / sizeof(uint16_t))
57 return -1;
58 }
59
60 return index;

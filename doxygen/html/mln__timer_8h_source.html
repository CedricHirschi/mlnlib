71uint8_t MLN_TIMER_GET_DIV(uint32_t period)
72{
73 uint8_t index = 0;
74
75 uint32_t max_period = mln_timer_max_periods[index] / MLN_TIMER_DIV;
76
77 while(period > max_period)
78 {
79 index++;
80
81 if(index == sizeof(mln_timer_divs) / sizeof(uint16_t)) return -1;
82
83 max_period = mln_timer_max_periods[index] / MLN_TIMER_DIV;
84 }
85
86 return index;
87}

69uint8_t MLN_TIMER_GET_DIV(uint32_t period)
70{
71 uint8_t index = 0;
72
73 while(period > mln_timer_max_periods[index] / (F_CPU / 1000000UL))
74 {
75 index++;
76
77 if(index == sizeof(mln_timer_divs) / sizeof(uint16_t)) return -1;
78 }
79
80 return index;
81}

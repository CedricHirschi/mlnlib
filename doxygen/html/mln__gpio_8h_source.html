115class mln_gpio
116{
117 volatile PORT_t *port;
118 volatile uint8_t *pin_conf;
119 uint8_t pin_num;
120 uint8_t pin_bm;
121
122public:
123 inline mln_gpio() = default;
124 inline mln_gpio(const PIN_t& new_pin, const PIN_DIR_t& dir, const bool inv = false)
125 {
126 port = new_pin.port;
127 pin_num = new_pin.pin_num;
128 pin_bm = (1 << pin_num);
129 pin_conf = (uint8_t *)port + 0x10 + pin_num;
130 *pin_conf = PORT_INVEN_bm * inv;
131
132 conf_dir(dir);
133 conf_pull(dir);
134 }
135
136 /*
137 * @brief Attach interrupt to pin
138 *
139 * @param dir Interrupt type
140 */
141 inline void attach_interrupt(PIN_ISC_t dir) { *pin_conf = (*pin_conf & 0xF8) | dir; }
142
143 /*
144 * @brief Set pin to HIGH
145 */
146 inline const void set(void) { port->OUTSET = pin_bm; }
147 /*
148 * @brief Clear pin to LOW
149 */
150 inline const void clear(void) { port->OUTCLR = pin_bm; }
151 /*
152 * @brief Toggle pin value
153 */
154 inline const void toggle(void) { port->OUTTGL = pin_bm; }
155 /*
156 * @brief Put value on pin
157 *
158 * @param val Value to put on pin
159 */
160 inline const void put(bool val) { (val ? port->OUTSET : port->OUTCLR) = pin_bm; }
161
162 /*
163 * @brief Get value of pin
164 */
165 inline const bool get(void) { return port->IN & pin_bm; }
166
167 /*
168 * @brief Configure direction of pin
169 *
170 * @param dir Direction to be set
171 */
172 inline void conf_dir(const PIN_DIR_t& dir)
173 {
174 if(dir == OUTPUT) port->DIR |= pin_bm;
175 else port->DIR &= ~pin_bm;
176
177 conf_pull(dir);
178 }
179 /*
180 * @brief Configure pull of pin
181 *
182 * @param pull Direction to be set
183 */
184 inline void conf_pull(const PIN_DIR_t& pull)
185 {
186 if(pull == INPUT_PULLUP) *pin_conf |= PORT_PULLUPEN_bm;
187 else *pin_conf &= ~PORT_PULLUPEN_bm;
188 }

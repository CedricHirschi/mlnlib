254 void init_pins(void)
255 {
256 if (setup->output)
257 init_pin(OPA_SETUP::PIN_OUT);
258
259 switch (setup->config)
260 {
261 case OPA_SETUP::TOPINS:
262 if (setup->pin_neg == OPA_SETUP::MUXNEG_INN)
263 init_pin(OPA_SETUP::PIN_INN);
264 if (setup->pin_pos == OPA_SETUP::MUXPOS_INP)
265 init_pin(OPA_SETUP::PIN_INP);
266 break;
267 case OPA_SETUP::FOLLOWER:
268 if (setup->pin_pos == OPA_SETUP::MUXPOS_INP)
269 init_pin(OPA_SETUP::PIN_INP);
270 break;
271 case OPA_SETUP::NONINV:
272 if (setup->pin_pos == OPA_SETUP::MUXPOS_INP)
273 init_pin(OPA_SETUP::PIN_INP);
274 break;
275 case OPA_SETUP::INV:
276 if (setup->pin_neg == OPA_SETUP::MUXNEG_INN)
277 init_pin(OPA_SETUP::PIN_INN);
278 break;

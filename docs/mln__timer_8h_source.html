73uint8_t MLN_TIMER_GET_DIV(uint32_t period)
74{
75 uint8_t index = 0;
76
77 uint32_t max_period = mln_timer_max_periods[index] / MLN_TIMER_DIV;
78
79 while (period > max_period)
80 {
81 index++;
82
83 if (index == sizeof(mln_timer_divs) / sizeof(uint16_t))
84 return -1;
85
86 max_period = mln_timer_max_periods[index] / MLN_TIMER_DIV;
87 }
88
89 return index;
90}

348 inline const void enable(const mln_opamp_device_t &opamp, const bool runstdby, const bool output, const bool lowpower)
349 {
350 // TIMEBASE: TIMEBASE(calculate)
351 OPAMP.TIMEBASE = MLN_OPAMP_TIMEBASE_US;
352 // PWRCTRL: IRSEL(lowpower)
353 OPAMP.PWRCTRL = (lowpower ? OPAMP_PWRCTRL_IRSEL_REDUCED_gc : OPAMP_PWRCTRL_IRSEL_FULL_gc);
354 // OPnCTRLA: ALWAYSON(true) and RUNSTDBY(runstdby) and OUTMODE(output)
355 switch (opamp)
356 {
357 case MLN_OPAMP_DEVICE_0:
358 OPAMP.OP0CTRLA = OPAMP_ALWAYSON_bm | (runstdby ? OPAMP_RUNSTBY_bm : 0) | (output ? OPAMP_OP1CTRLA_OUTMODE_NORMAL_gc : 0);
359 break;
360 case MLN_OPAMP_DEVICE_1:
361 OPAMP.OP1CTRLA = OPAMP_ALWAYSON_bm | (runstdby ? OPAMP_RUNSTBY_bm : 0) | (output ? OPAMP_OP1CTRLA_OUTMODE_NORMAL_gc : 0);
362 break;
363#ifdef MLN_OPAMP_DEVICE_2
364 case MLN_OPAMP_DEVICE_2:
365 OPAMP.OP2CTRLA = OPAMP_ALWAYSON_bm | (runstdby ? OPAMP_RUNSTBY_bm : 0) | (output ? OPAMP_OP2CTRLA_OUTMODE_NORMAL_gc : 0);
366 break;
367#endif
368 }
369 // CTRLA: ENABLE(true)
370 OPAMP.CTRLA = OPAMP_ENABLE_bm;
371 }
